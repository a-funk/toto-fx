Custom plugin

// A plugin is a plain object with categories and/or fx layers.
// Register with engine.use() before engine.init().

const flashPlugin = {
  name: 'flash',
  categories: {
    flash: {
      play(el, opts) {
        el.style.transition = 'filter 0.1s';
        el.style.filter = 'brightness(3)';
        setTimeout(() => {
          el.style.filter = '';
          if (opts.onDone) opts.onDone();
        }, 200);
      }
    }
  }
};

engine.use(flashPlugin);
engine.play('flash', someElement);
